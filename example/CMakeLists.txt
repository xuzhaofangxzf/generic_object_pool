project (generic_object_pool)

 -g
#  -rdynamic
 -fPIC
 -Wall
 -Wextra

 -Wpointer-arith

 -Wno-unused-parameter
 -Wno-reorder

 -DVALGRIND
 -D_FILE_OFFSET_BITS=64
 -DHAVE_STDINT_H
 -DHAVE_INTTYPES_H

STRING(REPLACE ";" " " CMAKE_CXX_FLAGS "${CXX_FLAGS}")

include_directories("../include")

add_executable(example1 ${srcs1})
